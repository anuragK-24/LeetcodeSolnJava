//easy simply solution
class Solution {
    public int singleNumber(int[] nums) {
        for(int i=0;i<nums.length;i++){
            int a=nums[i];
            int count=0;
            for(int j=0;j<nums.length;j++){
                if(nums[i]==nums[j]){
                    count++;
                } 
            }
            if(count!=2) {
                return nums[i];
            }
        }
        return 0;
    }
}
